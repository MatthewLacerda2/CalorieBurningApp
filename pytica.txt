import os

def list_files_and_folders(root_dir, depth=3, indent=0):
    if depth == 0:
        return

    for item in os.listdir(root_dir):
        full_path = os.path.join(root_dir, item)
        if os.path.isdir(full_path):
            if item != '.git' and 'client' not in item.lower():
                print("  " * indent + "- Folder:", item)
                list_files_and_folders(full_path, depth-1, indent+1)
        elif os.path.isfile(full_path):
            _, ext = os.path.splitext(item)
            if ext in ['.csproj', '.yml']:
                print("  " * indent + "  - File:", item)

root_directory = r"C:\Users\Matheus Lacerda\Desktop\Repos\CalorieBurningApp"
print("Root Directory:", root_directory)
list_files_and_folders(root_directory)
